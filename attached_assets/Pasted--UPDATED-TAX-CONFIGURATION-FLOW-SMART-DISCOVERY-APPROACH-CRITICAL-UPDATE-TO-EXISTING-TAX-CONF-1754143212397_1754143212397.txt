        askMultiStateQuestion();
    } else if (foundTaxes.includes('federal_income') && !foundTaxes.includes('fica_employer')) {
        askEmployerTaxQuestion();
    } else {
        // Skip to recommendations if no questions needed
        showTaxRecommendations();
    }
}

function askNextDiscoveryQuestion(questionType) {
    let questionText = '';
    let responseActions = [];
    
    switch (questionType) {
        case 'unemployment_handling':
            questionText = `I see you have employees but no unemployment tax setup in your documents.
            
❓ Do you handle federal and state unemployment taxes yourself, or does a payroll service (like ADP, Gusto, etc.) manage these for you?`;
            
            responseActions = [
                { action: 'handle-unemployment-myself', text: 'We handle it ourselves' },
                { action: 'payroll-service-handles', text: 'Payroll service handles this' },
                { action: 'not-sure-unemployment', text: 'Not sure / need to check' }
            ];
            break;
            
        case 'multistate_employees':
            questionText = `I notice you have employees in multiple states, but only found some state taxes.
            
❓ For your employees in other states - are they permanent staff or temporary/project workers?`;
            
            responseActions = [
                { action: 'permanent-multistate', text: 'Permanent employees' },
                { action: 'temporary-multistate', text: 'Temporary/project workers' },
                { action: 'mixed-multistate', text: 'Mix of both' }
            ];
            break;
            
        case 'employer_taxes':
            questionText = `I see federal income tax withholding but want to confirm employer obligations.
            
❓ Do you handle the employer portion of payroll taxes (Social Security/Medicare match) yourself?`;
            
            responseActions = [
                { action: 'handle-employer-taxes', text: 'Yes, we handle employer taxes' },
                { action: 'system-handles-employer', text: 'Our system handles this automatically' },
                { action: 'need-help-employer', text: 'Need help understanding this' }
            ];
            break;
    }
    
    addMessage(questionText, 'ai', responseActions);
    
    // Store current question context
    taxConfigurationState.currentQuestion = questionType;
}

// ADD this new function to handle discovery responses
function handleDiscoveryResponse(action) {
    const currentQuestion = taxConfigurationState.currentQuestion;
    let userAnswer = '';
    
    // Process the user's answer
    switch (action) {
        case 'handle-unemployment-myself':
            userAnswer = 'We handle it ourselves';
            taxConfigurationState.needsUnemploymentTax = true;
            break;
        case 'payroll-service-handles':
            userAnswer = 'Payroll service handles this';
            taxConfigurationState.needsUnemploymentTax = false;
            break;
        case 'permanent-multistate':
            userAnswer = 'Permanent employees';
            taxConfigurationState.needsMultiStateTax = true;
            break;
        case 'temporary-multistate':
            userAnswer = 'Temporary/project workers';
            taxConfigurationState.needsMultiStateTax = false;
            break;
        // ... handle other responses
    }
    
    // Add user message
    addMessage(userAnswer, 'user');
    
    // Move to recommendations with context
    setTimeout(() => {
        generateSmartRecommendations();
    }, 1000);
}

// ADD this new function
function generateSmartRecommendations() {
    const recommendations = [];
    const skipped = [];
    
    // Build recommendations based on discovery answers
    if (taxConfigurationState.needsUnemploymentTax) {
        recommendations.push({
            name: 'Federal FUTA',
            jurisdiction: 'IRS (Federal Government)',
            reason: 'Since you handle unemployment taxes yourself and have W-2 employees, federal unemployment insurance is required by law.',
            details: '0.6% rate on first $7,000 per employee annually, quarterly deposits'
        });
    } else {
        skipped.push({
            name: 'Federal FUTA',
            reason: 'Your payroll service handles this, so no separate setup needed.'
        });
    }
    
    // Show recommendations with reasoning
    let messageText = 'Thanks for clarifying! Based on your answers, here\'s what I recommend:\n\n';
    
    if (recommendations.length > 0) {
        messageText += '🎯 RECOMMENDED ADDITIONS:\n\n';
        recommendations.forEach(rec => {
            messageText += `✅ ${rec.name}\n→ ${rec.jurisdiction}\n💡 Why you need this: ${rec.reason}\n📋 Details: ${rec.details}\n\n`;
        });
    }
    
    if (skipped.length > 0) {
        messageText += '❌ NOT NEEDED:\n\n';
        skipped.forEach(skip => {
            messageText += `❌ ${skip.name}\n💡 Why skipping: ${skip.reason}\n\n`;
        });
    }
    
    messageText += 'Should I add the recommended taxes to your configuration?';
    
    addMessage(messageText, 'ai', [
        { action: 'add-recommended-taxes', text: 'Yes, add recommended taxes' },
        { action: 'review-recommendations', text: 'Let me review each one' },
        { action: 'skip-recommendations', text: 'Skip for now' }
    ]);
}
